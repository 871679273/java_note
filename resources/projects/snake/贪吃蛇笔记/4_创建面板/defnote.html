  6. /**
  7.  * @author : msb-zhaoss
  8.  * GamePanel本身不是一个面板
  9.  * 但是需要继承一个类：JPanel
  10.  * 那么GamePanel就具备面板的能力了
  11.  */
  12. public class GamePanel extends JPanel {
  13.     /*
  14.     paintComponent这个方法 就类似 我们平时写的main方法
  15.     它能做的事就是 ：底层也会直接调用，只要你写了这个方法，当需要画图的时候
  16.     底层就去自动调用这个方法
  17.     所以你想要画的东西，都要写在这个方法中
  18.      */
  19.     @Override
  20.     protected void paintComponent(Graphics g) {
  21.         super.paintComponent(g);
  22.         //先将面板的背景颜色画上：
  23.         this.setBackground(new Color(174, 207, 174));
  24.         //画一个图片：
  25.         /*
  26.         四个参数：this指代当前面板   g利用画笔   x,y指的就是举例 面板左上角的坐标
  27.          */
  28.         Images.headerImg.paintIcon(this,g,10,10);
  29.         //调节画笔的颜色：
  30.         g.setColor(new Color(190, 179, 201));
  31.         //画一个矩形：
  32.         /*
  33.         四个参数：x,y  设置坐标   后面两个参数：设置矩形的宽和高
  34.          */
  35.         g.fillRect(10,70,770,685);
  37.     }
  38. }

  1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  6. /**
  7.  * @author : msb-zhaoss
  8.  */
  9. public class StartGame {
  10.     //这是一个main方法，是程序的入口：
  11.     public static void main(String[] args) {
  12.         //创建一个窗体
  13.         JFrame jf = new JFrame();
  15.         //给窗体加一个标题
  16.         jf.setTitle("贪吃蛇小游戏by珊珊");
  17.         //获取屏幕的宽
  18.         int width = Toolkit.getDefaultToolkit().getScreenSize().width;
  19.         //获取屏幕的高：
  20.         int height = Toolkit.getDefaultToolkit().getScreenSize().height;
  21.         //设置窗体的位置：
  22.         jf.setBounds((width-800)/2,(height-800)/2,800,800);
  23.         //在关闭窗体的时候 程序也要停止：
  24.         jf.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
  25.         //进行设置，让窗体的大小不可变：
  26.         jf.setResizable(false);
  27.         //创建一个面板：
  28.         GamePanel gp = new GamePanel();
  29.         //将面板放入窗体中：
  30.         jf.add(gp);
  31.         //将窗体展现出来：--->必须放在最后
  32.         jf.setVisible(true);
  33.     }
  34. }
