1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  5. import java.awt.event.ActionEvent;
  6. import java.awt.event.ActionListener;
  7. import java.awt.event.KeyAdapter;
  8. import java.awt.event.KeyEvent;
  10. /**
  11.  * @author : msb-zhaoss
  12.  * GamePanel本身不是一个面板
  13.  * 但是需要继承一个类：JPanel
  14.  * 那么GamePanel就具备面板的能力了
  15.  */
  16. public class GamePanel extends JPanel {
  17.     //定义蛇的长度：
  18.     private int length;
  19.     //定义一个数组，专门存储蛇的x轴坐标：
  20.     int[] snakeX = new int[600];
  21.     //定义一个数组，专门存储蛇的y轴坐标：
  22.     int[] snakeY = new int[600];
  23.     //设置一个变量，定义小蛇的方向：
  24.     String direction = "R"; //方向： L   R  U  D
  25.     //定义一个变量，控制游戏是否开始：
  26.     boolean flag = false;
  27.     //创建一个定时器：
  28.     Timer timer;
  29.     //初始化方法：
  30.     public void init(){
  31.         length = 3;//蛇的初始化长度为3
  32.         //初始化蛇头的坐标：
  33.         snakeX[0] = 150;
  34.         snakeY[0] = 300;
  35.         //初始化第一节蛇身子坐标：
  36.         snakeX[1] = 125;
  37.         snakeY[1] = 300;
  38.         //初始化第二节蛇身子坐标：
  39.         snakeX[2] = 100;
  40.         snakeY[2] = 300;
  41.     }
  42.     public GamePanel(){
  43.         init();
  44.         //将焦点放在面板上：
  45.         this.setFocusable(true);
  46.         //加入一个键盘监听：
  47.         this.addKeyListener(new KeyAdapter() {
  48.             @Override
  49.             public void keyPressed(KeyEvent e) {
  50.                 super.keyPressed(e);
  51.                 int key = e.getKeyCode();
  52.                 if(key == 32){//监听到空格了
  53.                    /* System.out.println("你已经点击空格了");*/
  54.                     flag = !flag;
  55.                     //游戏如果是开始的，就变为暂停的
  56.                     //游戏如果是暂停的，就变为开始的
  57.                     repaint();//重新绘制
  58.                 }
  59.             }
  60.         });
  61.         //初始化定时器：创建定时器的对象
  62.         //参数两个：第一个参数设置：蛇多久动一次
  63.         //第二个参数：就是设置蛇具体怎么动-->重写了一个actionPerformed方法
  64.         //具体蛇头的逻辑要写在actionPerformed方法中
  65.         timer = new Timer(100, new ActionListener() {
  66.             @Override
  67.             public void actionPerformed(ActionEvent e) {
  68.                 if(flag){//如果游戏是开始的，蛇才会动
  69.                     //蛇的后一节 到 前一节位置上去：
  70.                     //实际就是改变蛇的坐标 -->改变的蛇身子
  71.                     for (int i = length-1;i>0;i--){
  72.                         snakeX[i] = snakeX[i-1];
  73.                         snakeY[i] = snakeY[i-1];
  74.                     }
  75.                     //改变蛇头：
  76.                     //往右动的话 蛇头 x坐标+25
  77.                     snakeX[0] = snakeX[0] + 25;
  79.                     //防止蛇飞出边界：
  80.                     if(snakeX[0] > 750){
  81.                         snakeX[0] = 25;
  82.                     }
  84.                     //蛇的坐标改完以后，重绘这个画面
  85.                     repaint();
  86.                 }
  87.             }
  88.         });
  89.         //定时器开启：
  90.         timer.start();
  91.     }
  95.     /*
  96.     paintComponent这个方法 就类似 我们平时写的main方法
  97.     它能做的事就是 ：底层也会直接调用，只要你写了这个方法，当需要画图的时候
  98.     底层就去自动调用这个方法
  99.     所以你想要画的东西，都要写在这个方法中
  100.      */
  101.     @Override
  102.     protected void paintComponent(Graphics g) {
  103.         super.paintComponent(g);
  104.         //先将面板的背景颜色画上：
  105.         this.setBackground(new Color(174, 207, 174));
  106.         //画一个图片：
  107.         /*
  108.         四个参数：this指代当前面板   g利用画笔   x,y指的就是举例 面板左上角的坐标
  109.          */
  110.         Images.headerImg.paintIcon(this,g,10,10);
  111.         //调节画笔的颜色：
  112.         g.setColor(new Color(190, 179, 201));
  113.         //画一个矩形：
  114.         /*
  115.         四个参数：x,y  设置坐标   后面两个参数：设置矩形的宽和高
  116.          */
  117.         g.fillRect(10,70,770,685);
  118.         //画蛇头：
  119.         if(direction.equals("R")){
  120.             Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  121.         }
  122.         if(direction.equals("L")){
  123.             Images.leftImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  124.         }
  125.         if(direction.equals("U")){
  126.             Images.upImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  127.         }
  128.         if(direction.equals("D")){
  129.             Images.downImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  130.         }
  131.         /*//画第一节身子：
  132.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  133.         //画第二节身子：
  134.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  135.         //将画身子的方法优化：
  136.         for (int i = 1;i<length;i++){
  137.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  138.         }
  140.         //判断游戏到底是开始还是暂停：
  141.         if(flag == false ){//游戏暂停：
  142.             //在面板上写文字：写游戏暂停的文字：
  143.             g.setColor(new Color(114, 98, 255));
  144.             //三个参数：字体样式，加粗效果，字号
  145.             g.setFont(new Font("微软雅黑",Font.BOLD,40));
  146.             //写文字
  147.             g.drawString("点击空格开始游戏！",250,330);
  148.         }/*else{//游戏开始
  149.             //什么都不用动。。不用管
  150.         }*/
  151.     }
  152. }
