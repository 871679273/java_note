1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  6. /**
  7.  * @author : msb-zhaoss
  8.  * GamePanel本身不是一个面板
  9.  * 但是需要继承一个类：JPanel
  10.  * 那么GamePanel就具备面板的能力了
  11.  */
  12. public class GamePanel extends JPanel {
  13.     //定义蛇的长度：
  14.     private int length;
  15.     //定义一个数组，专门存储蛇的x轴坐标：
  16.     int[] snakeX = new int[600];
  17.     //定义一个数组，专门存储蛇的y轴坐标：
  18.     int[] snakeY = new int[600];
  19.     //设置一个变量，定义小蛇的方向：
  20.     String direction = "R"; //方向： L   R  U  D
  21.     //初始化方法：
  22.     public void init(){
  23.         length = 3;//蛇的初始化长度为3
  24.         //初始化蛇头的坐标：
  25.         snakeX[0] = 150;
  26.         snakeY[0] = 300;
  27.         //初始化第一节蛇身子坐标：
  28.         snakeX[1] = 125;
  29.         snakeY[1] = 300;
  30.         //初始化第二节蛇身子坐标：
  31.         snakeX[2] = 100;
  32.         snakeY[2] = 300;
  33.     }
  34.     public GamePanel(){
  35.         init();
  36.     }
  40.     /*
  41.     paintComponent这个方法 就类似 我们平时写的main方法
  42.     它能做的事就是 ：底层也会直接调用，只要你写了这个方法，当需要画图的时候
  43.     底层就去自动调用这个方法
  44.     所以你想要画的东西，都要写在这个方法中
  45.      */
  46.     @Override
  47.     protected void paintComponent(Graphics g) {
  48.         super.paintComponent(g);
  49.         //先将面板的背景颜色画上：
  50.         this.setBackground(new Color(174, 207, 174));
  51.         //画一个图片：
  52.         /*
  53.         四个参数：this指代当前面板   g利用画笔   x,y指的就是举例 面板左上角的坐标
  54.          */
  55.         Images.headerImg.paintIcon(this,g,10,10);
  56.         //调节画笔的颜色：
  57.         g.setColor(new Color(190, 179, 201));
  58.         //画一个矩形：
  59.         /*
  60.         四个参数：x,y  设置坐标   后面两个参数：设置矩形的宽和高
  61.          */
  62.         g.fillRect(10,70,770,685);
  63.         //画蛇头：
  64.         if(direction.equals("R")){
  65.             Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  66.         }
  67.         if(direction.equals("L")){
  68.             Images.leftImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  69.         }
  70.         if(direction.equals("U")){
  71.             Images.upImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  72.         }
  73.         if(direction.equals("D")){
  74.             Images.downImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  75.         }
  76.         /*//画第一节身子：
  77.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  78.         //画第二节身子：
  79.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  80.         //将画身子的方法优化：
  81.         for (int i = 1;i<length;i++){
  82.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  83.         }
  85.     }
  86. }
