1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  5. import java.awt.event.ActionEvent;
  6. import java.awt.event.ActionListener;
  7. import java.awt.event.KeyAdapter;
  8. import java.awt.event.KeyEvent;
  9. import java.util.Random;
  11. /**
  12.  * @author : msb-zhaoss
  13.  * GamePanel本身不是一个面板
  14.  * 但是需要继承一个类：JPanel
  15.  * 那么GamePanel就具备面板的能力了
  16.  */
  17. public class GamePanel extends JPanel {
  18.     //定义蛇的长度：
  19.     private int length;
  20.     //定义一个数组，专门存储蛇的x轴坐标：
  21.     int[] snakeX = new int[600];
  22.     //定义一个数组，专门存储蛇的y轴坐标：
  23.     int[] snakeY = new int[600];
  24.     //设置食物的xy轴坐标：
  25.     int foodX ;
  26.     int foodY ;
  27.     //设置一个变量，定义小蛇的方向：
  28.     String direction = "R"; //方向： L   R  U  D
  29.     //定义一个变量，控制游戏是否开始：
  30.     boolean flag = false;
  31.     //创建一个定时器：
  32.     Timer timer;
  33.     //定义一个变量，专门用来接收积分：
  34.     int score ;
  35.     //初始化方法：
  36.     public void init(){
  37.         length = 3;//蛇的初始化长度为3
  38.         //初始化蛇头的坐标：
  39.         snakeX[0] = 150;
  40.         snakeY[0] = 300;
  41.         //初始化第一节蛇身子坐标：
  42.         snakeX[1] = 125;
  43.         snakeY[1] = 300;
  44.         //初始化第二节蛇身子坐标：
  45.         snakeX[2] = 100;
  46.         snakeY[2] = 300;
  47.         //初始化食物的位置：
  48.         foodX = 325;
  49.         foodY = 225;
  50.     }
  51.     public GamePanel(){
  52.         init();
  53.         //将焦点放在面板上：
  54.         this.setFocusable(true);
  55.         //加入一个键盘监听：
  56.         this.addKeyListener(new KeyAdapter() {
  57.             @Override
  58.             public void keyPressed(KeyEvent e) {
  59.                 super.keyPressed(e);
  60.                 int key = e.getKeyCode();
  61.                 System.out.println(key);
  62.                 if(key == KeyEvent.VK_SPACE){//监听到空格了
  63.                    /* System.out.println("你已经点击空格了");*/
  64.                     flag = !flag;
  65.                     //游戏如果是开始的，就变为暂停的
  66.                     //游戏如果是暂停的，就变为开始的
  67.                     repaint();//重新绘制
  68.                 }
  69.                 //上
  70.                 if(key == KeyEvent.VK_UP){
  71.                     direction = "U";
  72.                 }
  73.                 //下
  74.                 if(key == KeyEvent.VK_DOWN){
  75.                     direction = "D";
  76.                 }
  77.                 //左
  78.                 if(key == KeyEvent.VK_LEFT){
  79.                     direction = "L";
  80.                 }
  82.                 //右
  83.                 if(key == KeyEvent.VK_RIGHT){
  84.                     direction = "R";
  85.                 }
  86.             }
  87.         });
  88.         //初始化定时器：创建定时器的对象
  89.         //参数两个：第一个参数设置：蛇多久动一次
  90.         //第二个参数：就是设置蛇具体怎么动-->重写了一个actionPerformed方法
  91.         //具体蛇头的逻辑要写在actionPerformed方法中
  92.         timer = new Timer(100, new ActionListener() {
  93.             @Override
  94.             public void actionPerformed(ActionEvent e) {
  95.                 if(flag){//如果游戏是开始的，蛇才会动
  96.                     //蛇的后一节 到 前一节位置上去：
  97.                     //实际就是改变蛇的坐标 -->改变的蛇身子
  98.                     for (int i = length-1;i>0;i--){
  99.                         snakeX[i] = snakeX[i-1];
  100.                         snakeY[i] = snakeY[i-1];
  101.                     }
  102.                     //改变蛇头：
  103.                     //往右动的话 蛇头 x坐标+25
  104.                     if(direction.equals("R")){
  105.                         snakeX[0] = snakeX[0] + 25;
  106.                     }
  107.                     if(direction.equals("L")){
  108.                         snakeX[0] = snakeX[0] - 25;
  109.                     }
  110.                     if(direction.equals("U")){
  111.                         snakeY[0] = snakeY[0] - 25;
  112.                     }
  113.                     if(direction.equals("D")){
  114.                         snakeY[0] = snakeY[0] + 25;
  115.                     }
  118.                     //防止蛇飞出边界：
  119.                     //向右边界：
  120.                     if(snakeX[0] > 750){
  121.                         snakeX[0] = 25;
  122.                     }
  123.                     //向上边界：
  124.                     if(snakeY[0] < 100 ){
  125.                         snakeY[0] = 725;
  126.                     }
  128.                     //向左边界：
  129.                     if(snakeX[0] <25){
  130.                         snakeX[0] = 750;
  131.                     }
  133.                     //向下边界：
  134.                     if(snakeY[0] > 725){
  135.                         snakeY[0] = 100;
  136.                     }
  138.                     //当蛇头和食物的坐标重合的时候就是吃到食物了：
  139.                     if(snakeX[0] == foodX&&snakeY[0] == foodY){
  140.                         //蛇长度加1：
  141.                         length++;
  142.                         //改变食物的位置；-->重点：食物的坐标 无论是x  还是y必须是25的倍数
  143.                         foodX = 25 + 25*(new Random().nextInt(28));
  144.                         foodY = 100 + 25*(new Random().nextInt(25));
  145.                         //积分增长:
  146.                         score = (length-3)*10;
  147.                     }
  148.                     /*
  149.                     [25,725] --> [0,700]+25
  150.                     Math.random -- >  [0.0,1.0)
  151.                     Math.random*701 --> [0.0,700.0)
  152.                     (int)(Math.random*701)  -->[0,700]
  153.                     (int)(Math.random*701)+ 25
  155.                     [100,750] -->[0,650]+100
  156.                      */
  158.                     //蛇的坐标改完以后，重绘这个画面
  159.                     repaint();
  160.                 }
  161.             }
  162.         });
  163.         //定时器开启：
  164.         timer.start();
  165.     }
  169.     /*
  170.     paintComponent这个方法 就类似 我们平时写的main方法
  171.     它能做的事就是 ：底层也会直接调用，只要你写了这个方法，当需要画图的时候
  172.     底层就去自动调用这个方法
  173.     所以你想要画的东西，都要写在这个方法中
  174.      */
  175.     @Override
  176.     protected void paintComponent(Graphics g) {
  177.         super.paintComponent(g);
  178.         //先将面板的背景颜色画上：
  179.         this.setBackground(new Color(174, 207, 174));
  180.         //画一个图片：
  181.         /*
  182.         四个参数：this指代当前面板   g利用画笔   x,y指的就是举例 面板左上角的坐标
  183.          */
  184.         Images.headerImg.paintIcon(this,g,10,10);
  185.         //调节画笔的颜色：
  186.         g.setColor(new Color(190, 179, 201));
  187.         //画一个矩形：
  188.         /*
  189.         四个参数：x,y  设置坐标   后面两个参数：设置矩形的宽和高
  190.          */
  191.         g.fillRect(10,70,770,685);
  192.         //画蛇头：
  193.         if(direction.equals("R")){
  194.             Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  195.         }
  196.         if(direction.equals("L")){
  197.             Images.leftImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  198.         }
  199.         if(direction.equals("U")){
  200.             Images.upImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  201.         }
  202.         if(direction.equals("D")){
  203.             Images.downImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  204.         }
  205.         /*//画第一节身子：
  206.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  207.         //画第二节身子：
  208.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  209.         //将画身子的方法优化：
  210.         for (int i = 1;i<length;i++){
  211.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  212.         }
  213.         //画食物：
  214.         Images.foodImg.paintIcon(this,g,foodX,foodY);
  216.         //判断游戏到底是开始还是暂停：
  217.         if(flag == false ){//游戏暂停：
  218.             //在面板上写文字：写游戏暂停的文字：
  219.             g.setColor(new Color(114, 98, 255));
  220.             //三个参数：字体样式，加粗效果，字号
  221.             g.setFont(new Font("微软雅黑",Font.BOLD,40));
  222.             //写文字
  223.             g.drawString("点击空格开始游戏！",250,330);
  224.         }/*else{//游戏开始
  225.             //什么都不用动。。不用管
  226.         }*/
  228.         //画积分：
  230.         g.setColor(new Color(255, 248, 248));
  231.         //三个参数：字体样式，加粗效果，字号
  232.         g.setFont(new Font("微软雅黑",Font.BOLD,20));
  233.         //写文字
  234.         g.drawString("积分："+score,620,40);
  235.     }
  236. }
