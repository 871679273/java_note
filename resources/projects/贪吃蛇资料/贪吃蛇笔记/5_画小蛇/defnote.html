1. package com.msb.test01;
  3. import javax.swing.*;
  4. import java.awt.*;
  6. /**
  7.  * @author : msb-zhaoss
  8.  * GamePanel本身不是一个面板
  9.  * 但是需要继承一个类：JPanel
  10.  * 那么GamePanel就具备面板的能力了
  11.  */
  12. public class GamePanel extends JPanel {
  13.     //定义蛇的长度：
  14.     private int length;
  15.     //定义一个数组，专门存储蛇的x轴坐标：
  16.     int[] snakeX = new int[600];
  17.     //定义一个数组，专门存储蛇的y轴坐标：
  18.     int[] snakeY = new int[600];
  19.     //初始化方法：
  20.     public void init(){
  21.         length = 3;//蛇的初始化长度为3
  22.         //初始化蛇头的坐标：
  23.         snakeX[0] = 150;
  24.         snakeY[0] = 300;
  25.         //初始化第一节蛇身子坐标：
  26.         snakeX[1] = 125;
  27.         snakeY[1] = 300;
  28.         //初始化第二节蛇身子坐标：
  29.         snakeX[2] = 100;
  30.         snakeY[2] = 300;
  31.     }
  32.     public GamePanel(){
  33.         init();
  34.     }
  38.     /*
  39.     paintComponent这个方法 就类似 我们平时写的main方法
  40.     它能做的事就是 ：底层也会直接调用，只要你写了这个方法，当需要画图的时候
  41.     底层就去自动调用这个方法
  42.     所以你想要画的东西，都要写在这个方法中
  43.      */
  44.     @Override
  45.     protected void paintComponent(Graphics g) {
  46.         super.paintComponent(g);
  47.         //先将面板的背景颜色画上：
  48.         this.setBackground(new Color(174, 207, 174));
  49.         //画一个图片：
  50.         /*
  51.         四个参数：this指代当前面板   g利用画笔   x,y指的就是举例 面板左上角的坐标
  52.          */
  53.         Images.headerImg.paintIcon(this,g,10,10);
  54.         //调节画笔的颜色：
  55.         g.setColor(new Color(190, 179, 201));
  56.         //画一个矩形：
  57.         /*
  58.         四个参数：x,y  设置坐标   后面两个参数：设置矩形的宽和高
  59.          */
  60.         g.fillRect(10,70,770,685);
  61.         //画蛇头：
  62.         Images.rightImg.paintIcon(this,g,snakeX[0],snakeY[0]);
  63.         /*//画第一节身子：
  64.         Images.bodyImg.paintIcon(this,g,snakeX[1],snakeY[1]);
  65.         //画第二节身子：
  66.         Images.bodyImg.paintIcon(this,g,snakeX[2],snakeY[2]);*/
  67.         //将画身子的方法优化：
  68.         for (int i = 1;i<length;i++){
  69.             Images.bodyImg.paintIcon(this,g,snakeX[i],snakeY[i]);
  70.         }
  72.     }
  73. }
